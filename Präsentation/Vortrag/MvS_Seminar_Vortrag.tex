\begin{frame}
	\frametitle{ Das \textit{cheating husbands}-Rätsel }
	Induktionsrätsel zur Veranschaulichung von verteiltem Wissen
	\begin{itemize}
		\item Jede Ehefrau weiß, welche der anderen Ehefrauen betrogen wurden
		\item Keine Ehefrau weiß, ob sie selbst betrogen wurde
		\item Es gibt mindestens einen untreuen Ehemann
		\item Die Frauen dürfen nicht untereinander kommunizieren
		\item Findet eine Ehefrau heraus, dass sie betrogen wurde, erschießt sie ihren Ehemann um Mitternacht
	\end{itemize}
	
\end{frame}

\begin{frame}
	\frametitle{ Lösung des \textit{cheating husbands}-Rätsels }
	
	\begin{block}{Behauptung}
		Wenn n Ehemänner untreu waren, so werden diese alle in der n-ten Nacht erschossen.
	\end{block}
	
	Beweis mittels vollständiger Induktion über die Anzahl untreuer Ehemänner $n$:
	
	\begin{itemize}
		\item \textbf{Induktionsanfang $(n=1)$:} Die betrogene Ehefrau weiß von keinem untreuen Ehemann. Da es mindestens einen untreuen Ehemann gibt, ist sie betrogen worden. Die betrogene Frau erschießt ihren Ehemann in der ersten Nacht. 
	\end{itemize}
	
\end{frame}

\begin{frame}
	\frametitle{ Lösung des \textit{cheating husbands}-Rätsels }
	
	\begin{block}{Behauptung}
		Wenn n Ehemänner untreu waren, so werden diese alle in der n-ten Nacht erschossen.
	\end{block}
	 
	\begin{itemize}
		\item \textbf{Induktionsvoraussetzung:} Die Behauptung gelte für den Fall mit $n$ untreuen Ehemännern.   
	\end{itemize}
	 
	\begin{itemize}
		\item \textbf{Induktionsschritt $(n+1)$:} Betrogene Ehefrauen wissen von $n$ untreuen Ehemännern. Wenn es nur $n$ untreue Ehemänner gäbe, würden sie nach Induktionsvoraussetzung in der n-ten Nacht erschossen werden. In der n-ten Nacht fallen keine Schüsse, sodass die betrogenen Frauen dann wissen, dass sie selbst betrogen wurden.
	\end{itemize}
	
	\begin{flushright}
		$\square$
	\end{flushright}
\end{frame}

\section{Logik des Wissens}

\begin{frame}
	\frametitle{ Logikoperatoren des Wissens }
	
	\begin{block}<1->{mögliche Welten}
		Zustände des Systems, die mit dem lokalen Zustand des Prozesses verträglich sind.
	\end{block}
	
	\begin{block}<2->{Kripke-Modell M}
		Ein Kripke-Modell $M$ zu einem verteilten System mit $n$ Prozessen und einer Menge atomarer Aussagen $\Phi$ ist ein Tupel $(S,\pi,\mathcal{K}_1,...,\mathcal{K}_n)$.
	\end{block}
	\begin{itemize}
		\item<2-> $S$: Alle möglichen Systemzuständen
		\item<2-> $\pi$: Belegung, die jeder atomaren Aussage einen Wahrheitswert zuordnet $\forall s\in S, \pi(s):\Phi \rightarrow \{0,1\}$
		\item<2-> $\mathcal{K}_i$: Binäre Relation auf $S$, die jedem Prozess $i$ seine möglichen Welten zuordnet
	\end{itemize}
	
\end{frame}

\begin{frame}
	\frametitle{ Logikoperatoren des Wissens }
	
	Epistemische Logikoperatoren erweitern die Aussagenlogik\bigskip
	\begin{block}{Wissensoperatoren}
		\begin{itemize}
			\item<1-> (M,s) $\vDash \phi$, falls $\pi(s)(\phi) = true$
			\item<2-> (M,s) $\vDash K_i(\phi)$, falls für alle t mit $(s,t) \in \mathcal{K}_i $ gilt:\\ (M,t) $\vDash \phi$
			\item<3-> (M,s) $\vDash E^1(\phi)$, falls (M,s) $\vDash\land_{i\in N}K_i(\phi)$
			\item<4-> (M,s) $\vDash E^{k+1}(\phi)$ mit $k\ge 1$, falls (M,s) $\vDash\land_{i\in N}K_i(E^k(\phi))$
			\item<5-> (M,s) $\vDash C(\phi)$, falls (M,s) $\vDash\land_{k\in Z^*}E^k(\phi)$
		\end{itemize}
	\end{block}
	
	\begin{itemize}
		\item $ K_i(\phi)$: Prozess i weiß, dass $\phi$ gilt.
		\item $E^1(\phi)$: Jeder weiß, dass $\phi$ gilt.
		\item $E^2(\phi)$: Jeder weiß, dass jeder weiß, dass $\phi$ gilt.
		\item $C(\phi)$: Es ist gemeinsames Wissen, dass $\phi$ gilt.
	\end{itemize}
	
\end{frame}

\section{Wissen in synchronen und asynchronen Systemen}

\begin{frame}
	\frametitle{ Wissen in asynchronen Systemen}
	
	\begin{itemize}
		\item asynchrone Systeme :
		\begin{block}{\textit{cheating-Husband}-Erweiterung}
			\begin{itemize}
				\item Königin Henrietta II beerbt Henrietta I
				\item Einführung des Mail-Systems
				\item Jeder Brief der versandt wird \textit{eventuell} sein Ziel erreichen
			\end{itemize}
		\end{block}
	\end{itemize}
\end{frame}
\begin{frame}
	\frametitle{ Wissen in asynchronen Systemen}
	
		\begin{block}{Satz}
			Wenn es mehr wie einen untreuen Ehemann gibt und dieses Nachricht wird über einen asynchronen Kanal verbreitet, so wird kein Ehemann erschossen.
		\end{block}
		
		\begin{block}{Beweis (logisch)}
			\begin{itemize}
				\item Durch asynchrones Versenden gibt es ein \textit{eventuelles} Gemeinsames Wissen
				\item Der Brief der Königin erreicht unter Umständen ihr Ziel
				\item Die Adressatin des Briefes kann nicht sicher sein, dass andere Ehefrauen ebenfalls einen Brief erhalten haben
				\item So kann keine Frau aus \textit{ruhigen} Nächten schließen, dass Ihr Ehemann der Betrüger ist.
				
				\item Ähnlichkeit => \textit{Problem der Byzantinischen Generäle}
			\end{itemize}
		\end{block}
\end{frame}

\begin{frame}
	\frametitle{ Wissen in synchronen Systemen}
	
	\begin{itemize}
		\item schwache synchrone Systeme (I)
		\begin{block}{\textit{cheating-Husband}-Erweiterung}
			\begin{itemize}
				\item Königin Henrietta III beerbt Henrietta II
				\item Upgrade des Mail-Systems
				\item Jeder Brief soll nach Versand maximal 1 Tag benötigten zum Ziel 
			\end{itemize}
		\end{block}	
		
		\begin{block}{b-common-knowledge}
			Innerhalb von b Tagen erhält jede Frau einen Brief und weiß, dass innerhalb von b Tagen jede Frau über das gleiche Wissen verfügt.
		\end{block}
	\end{itemize}
\end{frame}
\begin{frame}
	\frametitle{ Wissen in synchronen Systemen}
	
	\begin{itemize}

\section{Varianten des gemeinsamen Wissens}

\begin{frame}
	\frametitle{ Übersicht - Gemeinsames Wissen}
	\begin{itemize}
		\item Epsilon gemeinsames Wissen
		\item Eventuelles gemeinsames Wissen
		\item Zeitabhängiges gemeinsames Wissen
		\item Gleichzeitiges gemeinsames Wissen
	\end{itemize}
	
\end{frame}
\begin{frame}
	\frametitle{ Epsilon gemeinsames Wissen }
	
	\begin{block}{Beschreibung}
		Die Prozesse treffen eine Vereinbarung innerhalb von $\epsilon$ Zeiteinheiten. Es gilt jedoch zu beachten, dass es in einem asynchronen System keine exakten Zeiten geben kann.
	\end{block}	
		\begin{block}{Definition}
		\begin{itemize}
			\item $E^\epsilon$, jeder Prozess erhält innherhalb von $\epsilon$ Zeiteinheiten die Nachricht
			\item $C^\epsilon(\phi) $, größter Fixpunkt von X
			\item $X=E^\epsilon(\phi \wedge X) $, X ist die freie Variable im Fixpunkt-Operator 
			%\widehat{=} => entspricht-Zeichen
		\end{itemize}
	\end{block}
	
\end{frame}
\begin{frame}
	\frametitle{ Eventuelles gemeinsames Wissen }
	
	\begin{block}{Beschreibung}
	Die Prozesse treffen eine Vereinbarung in einem globalen Status der Ausführung. Diese kann, aber muss nicht konsistent sein.
	\end{block}	
	\begin{block}{Definition}
	\begin{itemize}
			\item $E^\Diamond$, jeder Prozess hat eventuell das Wissen (zu einem Zeipunkt der Ausführung)
			\item $C^\Diamond(\phi) $, größter Fixpunkt von X
			\item $X=E^\Diamond(\phi \wedge X) $

		\end{itemize}
\end{block}
\end{frame}

\begin{frame}
	\frametitle{ Zeitabhängiges gemeinsames Wissen }
	
	\begin{block}{Beschreibung}
		Prozesse erreichen Vereinbarungen mit einem Status, der die gleiche lokale Zeit hat. 
		(Sind alle Uhren immer perfekt synchronisiert, so entspricht das Ergebnis regulär dem gemeinsamen Wissen.)
	\end{block}	
	\begin{block}{Definition}
	\begin{itemize}
			\item $K^T_i(\phi)$, Prozess \textit{i} weiß $\phi$ zum lokalen Zeitpunk T
			\item $E^T(\phi) = \bigwedge_i K^T_i(\phi) $ und $C^T(\phi)$, größter Fixpunkt von X
			\item $X=E^T(\phi \wedge X)$

		\end{itemize}
\end{block}
\end{frame}
\begin{frame}
	\frametitle{ Gleichzeitiges gemeinsames Wissen }
	
		\begin{block}{Beschreibung}
		Prozesse erreichen lokal einen konsistenten Status, welcher zu einem konsistenten Schnittpunkt gehört. Wenn Prozess $P_i$ gemeinsames Wissen von $\phi$ erlangen will, so hat Prozess $P_j$ das selbe wissen in einem Konsistenten Status.
	\end{block}	

\end{frame}
\begin{frame}
\frametitle{ Gleichzeitiges gemeinsames Wissen }
	\begin{block}{Definition}
	\begin{itemize}
			\item (a,c)	$\models$ $\phi$, wenn Schnittpunk c von Ausführung a = TRUE
			\item (a,c)	$\models$ $K_i(\phi)$, wenn $\forall(a',c')$, $((a',c') \sim_i (a,c) \implies (a',c') \models \phi)$
			\item (a,c)	$\models$ $P_i(\phi)$, wenn $\exists(a,c')$, $((a,c') \sim_i (a,c) \wedge (a,c') \models \phi)$
			\item (a,c)	$\models$ $E^{C^0}(\phi)$, wenn $(a,c) \models \phi$
			\item (a,c)	$\models$ $E^{C^1}(\phi)$, wenn $(a,c) \models \bigwedge_{i\epsilon N}\; K_i P_i(\phi)$
			\item (a,c)	$\models$ $E^{C^k+1}(\phi)$, wenn $(a,c) \models \bigwedge_{i\epsilon N}\; K_i P_i(E^{C^k}(\phi))$ für $k \geq 1$
			\item (a,c)	$\models$ $C^C(\phi)$, wenn $(a,c) \models X = E^C(X \wedge \phi)$ 
			$C^c \implies \bigwedge_{k\epsilon Z*}(E^c)^k(\phi) $ 
		\end{itemize}
\end{block}
\end{frame}

\section{Zusätzliche Kommunikation}

\begin{frame}
	\frametitle{ Zusätzliche Kommunikation }
	\begin{block}{Problem}
		In einem verteilten System soll das Konsens-Problem gelöst werden. Die Prozesse haben einen gemeinsamen Speicher von einem Bit. Es gibt maximal zwei verschiedene Werte im System, diese unterscheiden sich nur um eins.
	\end{block}
	Dies entspricht dem \textit{cheating husbands}-Rätsel mit folgender Erweiterung:
	%Zwei Werte: Frauen die nicht betrogen wurden kennen alle, die Betrogenen kennen einen weniger. Müssen sich darauf einigen, welche Anzahl untreuen Ehemänner die tatsächliche ist.
	\begin{itemize}
		\item Die Ehefrauen dürfen um Mitternacht in die Luft schießen
	\end{itemize}
		
\end{frame}

\begin{frame}
	\frametitle{ Lösung des Problems }
	Folgendes Kommunikationsprotokoll löst das Problem:
	\begin{itemize}
		\item 1. Falls eine Ehefrau von $\mathbf{k_1}$ (mit $k_1 \text{ mod } 3 = 1$) untreuen Ehemännern weiß, schießt sie in der ersten Nacht in die Luft.
		\item 2. Falls eine Ehefrau von $\mathbf{k_2}$ (mit $k_2 \text{ mod } 3 = 2$) untreuen Ehemännern weiß und \textbf{in der ersten Nacht keine Schüsse} gefallen sind, erschießt sie ihren Ehemann in der zweiten Nacht.
		\item 3. Falls eine Ehefrau von $\mathbf{k_0}$ (mit $k_0 \text{ mod } 3 = 0$) untreuen Ehemännern weiß und \textbf{in der ersten Nacht Schüsse} gefallen sind, erschießt sie ihren Ehemann in der zweiten Nacht.
	\end{itemize}
	
\end{frame}

\begin{frame}
	\frametitle{ Lösung des Problems }
	\begin{itemize}
		\item 4. Falls \textbf{in den ersten beiden Nächten keine Schüsse} gefallen sind und $\mathbf{k_0 \ne0}$, erschießen alle Ehefrauen ihren Mann in der dritten Nacht.
		\item 5. Falls \textbf{in der ersten Nacht Schüsse} gefallen sind, \textbf{in der zweiten Nach keine Schüsse} zu hören waren, erschießen alle Ehefrauen, die in der ersten Nacht bereits geschossen haben, ihren Mann in der dritten Nacht. 
	\end{itemize}
	
	\begin{block}{Satz}
		Das Protokoll löst das Problem in maximal drei Nächten. Es gibt kein Protokoll, das weniger als drei Nächte benötigt.
	\end{block}
	
\end{frame}
